	EXPORT START

	MOV R1, #0  
	MOV R2, #05  
	MOV R3, #04

	MOVS R3, #0

	MVN R2, R2  
STOP B STOP  
	END

	AREA PROG1,CODE, READONLY 
ENTRY  
	EXPORT START  
START  
	LDR R0,=10  
	LDR R1,=20  
	ADD R2,R0,R1  
STOP B STOP  
	END

3A)ADD FIRST 10 NUMBER

	AREA PROG2, CODE, READONLY  
ENTRY  
	EXPORT START  
START  
	LDR R0, =10  
	LDR R1, =0  
	LDR R2, =1  
LOOP  
	ADD R1, R2, R1  
	ADD R2, R2, #1  
	SUBS R4, R0, R2  
	BNE LOOP  
	ADD R1, R2, R1  
STOP B STOP  
	END 
4A)REV SUB OF TWO 32BIT NUMBER

	AREA MYCODE, CODE, READONLY  
ENTRY  
	EXPORT START  
START  
	LDR R1, =0X11111111  
	LDR R2, =0X22222222  
	RSB R0, R1, R2  
STOP B STOP  
	END 

5A)FACTORIAL OF NUMBER

	AREA PROG5, CODE, READONLY  
ENTRY  
	EXPORT START 
START  
	MOV R6, #10  
	MOV R7, #1  
LOOP CMP R6, #0  
	MULGT R7, R6, R7  
	SUBGT R6, R6, #1  
	BGT LOOP  
STOP B STOP  
	END
